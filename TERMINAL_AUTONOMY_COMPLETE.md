## 🔧 快速开始

### 基本使用

```python
from core.autonomy_manager import get_autonomy_manager

# 初始化
manager = get_autonomy_manager()
manager.initialize()

# 手动改进
result = await manager.manual_improvement(max_fixes=10)
print(f"发现: {result['problems_found']}, 修复: {result['fixes_successful']}")

# 自动改进
manager.enable_auto_improvement(interval=300)

# 查看学习报告
from core.knowledge_integration import get_knowledge_integration
knowledge = get_knowledge_integration()
report = knowledge.generate_learning_report()
print(report)

# 关闭
manager.shutdown()
```

### 高级使用

```python
from core.autonomy_manager import get_autonomy_manager
from core.knowledge_integration import get_knowledge_integration

# 初始化
manager = get_autonomy_manager()
manager.initialize()
knowledge = get_knowledge_integration()

# 启用自动改进
manager.enable_auto_improvement(interval=300)

# 让系统运行一段时间...
# 弥娅会自动：
# 1. 扫描问题
# 2. 评估风险
# 3. 做出决策
# 4. 执行修复
# 5. 记录结果
# 6. 学习模式

# 8. 持续优化

# 查看状态
status = manager.get_status()
print(f"总决策: {status['engine']['total_decisions']}")
print(f"成功修复: {status['engine']['successful_fixes']}")

# 查看学习报告
report = knowledge.generate_learning_report()
print(report)

# 分析成功模式
success_analysis = knowledge.analyze_success_patterns()
print(f"高成功率模式: {success_analysis['total_high_success']}")

# 关闭
manager.shutdown()
knowledge.save_all()
```
